import java.util.*;

class BackgroundJobExecution {
    public static void main(String[] args) {
        Runnable task1 = () -> System.out.println("Task 1 is running in background");
        Runnable task2 = () -> System.out.println("Task 2 is running in background");

        Thread thread1 = new Thread(task1);
        Thread thread2 = new Thread(task2);

        thread1.start();
        thread2.start();
    }
}
